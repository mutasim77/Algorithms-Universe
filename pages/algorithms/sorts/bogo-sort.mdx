## What's Bogosort? ğŸ’¡
BogoSort or "Monkey Sort" is an impractical and humorous sorting algorithm.
It's often humorously referred to as a "stupid sort" because it relies on pure luck to sort the elements.
BogoSort takes a novel approach to sorting by randomly permuting the elements in an array until it stumbles 
upon a permutation where the array is sorted. It relies solely on random chance and can take an indefinite 
amount of time to complete, making it one of the most inefficient sorting algorithms.

function isSorted(arr) {
  for (let i = 1; i < arr.length; i++) {
    if (arr[i] < arr[i - 1]) {
      return false;
    }
  }
  return true;
}

function bogosort(arr) {
  while (!isSorted(arr)) {
    arr = shuffle(arr);
  }
  return arr;
}

function shuffle(arr) {
  for (let i = arr.length - 1; i > 0; i--) {
    const j = Math.floor(Math.random() * (i + 1));
    [arr[i], arr[j]] = [arr[j], arr[i]];
  }
  return arr;
}
